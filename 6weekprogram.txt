    function updateSummary() {
        const workout = programData[currentWeek][currentDay];
        if (workout.isRestDay) return;
        
        let totalSets = 0;
        let completedSets = 0;
        let totalVolume = 0;
        
        workout.exercises.forEach((exercise, exerciseIndex) => {
            for (let set = 1; set <= exercise.sets; set++) {
                totalSets++;
                const weightInput = document.querySelector(`input[data-exercise="${exerciseIndex}"][data-set="${set}"][data-field="weight"]`);
                const repsInput = document.querySelector(`input[data-exercise="${exerciseIndex}"][data-set="${set}"][data-field="reps"]`);
                const completedInput = document.querySelector(`input[data-exercise="${exerciseIndex}"][data-set="${set}"][data-field="completed"]`);
                
                if (completedInput && completedInput.checked) {
                    completedSets++;
                }
                
                const weight = parseFloat(weightInput?.value) || 0;
                const reps = parseFloat(repsInput?.value) || 0;
                totalVolume += weight * reps;
            }
        });
        
        const startTime = document.getElementById('startTime')?.value;
        const endTime = document.getElementById('endTime')?.value;
        let duration = 0;
        
        if (startTime && endTime) {
            const start = new Date(`2000-01-01T${startTime}`);
            const end = new Date(`2000-01-01T${endTime}`);
            duration = Math.round((end - start) / 60000);
        }
        
        document.getElementById('totalSets').textContent = totalSets;
        document.getElementById('completedSets').textContent = completedSets;
        document.getElementById('totalVolume').textContent = totalVolume.toLocaleString();
        document.getElementById('workoutDuration').textContent = duration > 0 ? duration : '-';
    }
    
    function updateWeekProgress() {
        const container = document.getElementById('weekProgress');
        let html = '';
        
        for (let week = 1; week <= 6; week++) {
            const weekLabel = ['Foundation', 'Build', 'Intensity', 'Deload', 'Peak', 'Max Effort'][week - 1];
            const isCompleted = isWeekCompleted(week);
            const isActive = week === currentWeek;
            
            html += `
                <div class="week-card ${isCompleted ? 'completed' : ''} ${isActive ? 'active' : ''}" onclick="selectWeek(${week})">
                    <div class="week-number">W${week}</div>
                    <div class="week-label">${weekLabel}</div>
                </div>
            `;
        }
        
        container.innerHTML = html;
    }
    
    function isWeekCompleted(week) {
        // Check if all non-rest days have data
        for (let day = 1; day <= 6; day++) {
            const workout = programData[week][day];
            if (workout && !workout.isRestDay) {
                const logKey = `w${week}d${day}`;
                if (!workoutLogs[logKey] || !workoutLogs[logKey].startTime) {
                    return false;
                }
            }
        }
        return true;
    }

    function saveWorkout() {
        const logKey = `w${currentWeek}d${currentDay}`;
        const workout = programData[currentWeek][currentDay];
        
        if (workout.isRestDay) {
            workoutLogs[logKey] = {
                activity1: document.getElementById('rest1')?.checked,
                activity2: document.getElementById('rest2')?.checked,
                activity3: document.getElementById('rest3')?.checked,
                activity4: document.getElementById('rest4')?.checked,
                notes: document.getElementById('recoveryNotes')?.value,
                date: document.getElementById('workoutDate').value
            };
        } else {
            const log = {
                startTime: document.getElementById('startTime')?.value,
                endTime: document.getElementById('endTime')?.value,
                feeling: document.getElementById('feeling')?.value,
                date: document.getElementById('workoutDate').value,
                week: currentWeek,
                day: currentDay
            };
            
            workout.exercises.forEach((exercise, exerciseIndex) => {
                const exerciseLog = {};
                
                for (let set = 1; set <= exercise.sets; set++) {
                    const weight = document.querySelector(`input[data-exercise="${exerciseIndex}"][data-set="${set}"][data-field="weight"]`)?.value;
                    const reps = document.querySelector(`input[data-exercise="${exerciseIndex}"][data-set="${set}"][data-field="reps"]`)?.value;
                    const rest = document.querySelector(`input[data-exercise="${exerciseIndex}"][data-set="${set}"][data-field="rest"]`)?.value;
                    const completed = document.querySelector(`input[data-exercise="${exerciseIndex}"][data-set="${set}"][data-field="completed"]`)?.checked;
                    
                    exerciseLog[`set${set}`] = { weight, reps, rest, completed };
                }
                
                const notes = document.querySelector(`textarea[data-exercise="${exerciseIndex}"][data-field="notes"]`)?.value;
                exerciseLog.notes = notes;
                
                log[`exercise${exerciseIndex}`] = exerciseLog;
            });
            
            workoutLogs[logKey] = log;
        }
        
        localStorage.setItem('6weekLogs', JSON.stringify(workoutLogs));
        updateWeekProgress();
        
        const notification = document.getElementById('saveNotification');
        notification.classList.add('show-notification');
        setTimeout(() => {
            notification.classList.remove('show-notification');
        }, 2000);
    }

    function populateProgressCharts() {
        const exercises = {
            bench: { name: 'Barbell Bench Press', chart: 'benchChart' },
            squat: { name: 'Barbell Back Squat', chart: 'squatChart' },
            deadlift: { name: 'Barbell Deadlift', chart: 'deadliftChart' },

        };
        
        Object.values(exercises).forEach(ex => {

            let html = '';
            
            for (let week = 1; week <= 6; week++) {
                const data = getExerciseData(week, ex.name);
                html += `
                    <tr>
                        <td>Week ${week}</td>
                        <td>${data.weight || '-'}</td>
                        <td>${data.reps || '-'}</td>
                    </tr>
                `;
            }
            
            tbody.innerHTML = html;
        });
    }
    
    function getExerciseData(week, exerciseName) {
        // Find exercise in week data
        for (let day = 1; day <= 6; day++) {
            const workout = programData[week][day];
            if (!workout || workout.isRestDay) continue;
            
            const exerciseIndex = workout.exercises.findIndex(ex => ex.name === exerciseName);
            if (exerciseIndex === -1) continue;
            
            const logKey = `w${week}d${day}`;
            const log = workoutLogs[logKey];
            if (!log) continue;
            
            const exerciseLog = log[`exercise${exerciseIndex}`];
            if (!exerciseLog) continue;
            
            // Get first set data
            const set1 = exerciseLog.set1;
            if (set1 && set1.weight && set1.reps) {
                return { weight: set1.weight, reps: set1.reps };
            }
        }
        
        return { weight: null, reps: null };
    }
    
    function exportData() {

        const dataBlob = new Blob([dataStr], { type: 'application/json' });
        const url = URL.createObjectURL(dataBlob);
        const link = document.createElement('a');
        link.href = url;
        link.download = '6-week-training-data.json';
        link.click();
    }
